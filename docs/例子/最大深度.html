最大深度

最大深度

package main

import (
    "fmt"

    "github.com/gocolly/colly"
)

func main() {
    // 创建默认的收集器
    c := colly.NewCollector(
        // 最大深度为1，代表只有在爬取页面上的链接才会被访问，之后页面的链接不会被访问
        colly.MaxDepth(1),
    )

    // 处理具有href属性的a标签
    c.OnHTML("a[href]", func(e *colly.HTMLElement) {
        link := e.Attr("href")
        // 打印链接
        fmt.Println(link)
        // 访问这个链接
        e.Request.Visit(link)
    })

    // 开始爬取https://en.wikipedia.org
    c.Visit("https://en.wikipedia.org/")
}
